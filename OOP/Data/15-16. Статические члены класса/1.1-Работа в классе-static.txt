#include <iostream>
#include <time.h>

#define RAND(Min,Max)  (rand()%((Max)-(Min)+1)+(Min))
struct LC { LC() { system("chcp 1251 > nul"); srand(time(0)); }~LC() { cin.get(); cin.get(); } } _;

// пртотип переменной,
// объявление внешнего глобального имени
extern int a;
int a = 300;

int func1() {
  int a = 0; // временная локальная переменная
  a++;
  return a;
}

int func2() {
  a++;
  return a;
}

int func3() {
  static int a = 10; // статическая переменная
  a++;

}

{

    cout << func1() << "\t"
