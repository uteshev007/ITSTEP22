Функции-члены для подсчета площади должны быть

*/

class Area {
private:
  float S; // площадь последней посчитанной фигуры
public:
  float Circle(float R) {
    S = M_PI * R * R;
    return S;
  }
  float Square(float A) {
    return S = A * A;
  }
  void Print() {
    cout << "S = " << S << endl;
  }

}; // class Area
