//#define RND(Min,Max) ( rand()%((Max)-(Min)+1)+(Min) )

// random_device - класс для генерации настоящих случ. чисел
//  генерируется микропроцессором
//  диапазон: 0 .. 4'294'967'295
random_device rd; // rd - переменная типа random_device
#define RND(Min,Max) ( rd()%((Max)-(Min)+1)+(Min) )

// Генератор псевдослучайных чисел
// Твистер Мерсена, Вихрь Мерсена 
// дапазон: 0 .. 18 446 744 073 709 551 616 (20 разрядов)
mt19937_64 mt; // mt - переменная типа mt19937_64

int main()
{
  for (int i = 0; i < 10; i++) {
    // rd() - функция возвращает unsigned int
    //  случ. 0 .. 4'294'967'295
    //cout << rd() << endl;
    cout << RND(0, 87654321) << endl;
  }
  cout << endl;
  // srand(time(0));
  //mt.seed(time(0));

  mt.seed(rd());
  for (int i = 0; i < 10; i++) {
    //cout << mt() << endl;
    // [20, 100]
    cout << mt() % (100 -20+1)+20 << endl;
  }
}
