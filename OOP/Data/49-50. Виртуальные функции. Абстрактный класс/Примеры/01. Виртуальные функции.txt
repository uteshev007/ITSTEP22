#include <iostream>
using namespace std;

class Weapon
{
public:
	//тут будут данные-члены, которыми может описываться, например, как 
	//толщина дубины, так и количество гранат в гранатомете
	//эта часть для нас не важна

	void Use1() { cout << "ЛКМ общее\n"; }//обычно - левая кнопка мыши
	void Use2() { cout << "ПКМ общее\n"; }//обычно - правая кнопка мыши

	 //тут будут еще какие-то данные-члены и методы
};

class knife : public Weapon
{
	void Use1() { cout << "ЛКМ для ножа\n"; }//обычно - левая кнопка мыши
	void Use2() { cout << "ПКМ для ножа\n"; }//обычно - правая кнопка мыши
};

class club : public Weapon
{
	void Use1() { cout << "ЛКМ для дубины\n"; }//обычно - левая кнопка мыши
	void Use2() { cout << "ПКМ для дубины\n"; }//обычно - правая кнопка мыши
};


int main() {
	setlocale(LC_ALL, "RUSSIAN");

	Weapon *Arms[10]; //массив указателей на объекты типа Weapon
	Arms[0] = new knife;
	Arms[1] = new club;
	int TypeOfWeapon = 1;

	if ("нажата левая кнопка мыши") Arms[TypeOfWeapon]->Use1();
	else Arms[TypeOfWeapon]->Use2();

	cin.get();
}
