template <class T>
class Stack
{
private:
  Data;
  Size, Depth;
public:
  Stack();
  Stack(int depth);
  Stack(const Stack& obj);
  Stack& operator=(const Stack& obj);
  ~Stack();

  int GetSize();
  int GetDepth();

  void SetDepth(int newDepth);

  void push(T val);
  T pop();
  T top();
  T& top();

  friend ostream& operator << (ostream& os, const Stack<T>& obj);
}; // class Stack<>;
