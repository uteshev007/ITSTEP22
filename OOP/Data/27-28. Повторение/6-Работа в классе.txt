/*
39. *Написать программу, которая вычисляет сумму
элементов массива, используя ДВА указателя на
массив целых чисел.
Первый указатель двигается с начала массива,
второй - с конца.
*/
int sum1(int arr[], int size)
{
  int sum = 0;
  int* p1 = arr, * p2 = &arr[size-1];
  for (int i = 0; i < size / 2; i++) {
    sum += *p1;
    sum += *p2;
    p1++;
    p2--;
  }
  if (p1 == p2) { sum += *p1; }
  return sum;
}

int sum2(int arr[], int size)
{
  int sum = 0;
  int* p1 = arr, * p2 = &arr[size - 1];
  while (p1 < p2) {
    sum += *p1++ + *p2--;
  }
  if (p1 == p2) { sum += *p1; }
  return sum;
}

int main()
{
  int n;
  cout << "Enter size of array - ";
  cin >> n;
  int* arr = new int[n];
  for (int i = 0; i < n; i++) {
    arr[i] = RAND(1, 9);
    cout << arr[i] << "  ";

  cout << endl << "I--------I" << endl;
  cout << "Sum of array = " << sum1(arr, n) << endl;

  cout << "Sum of array = " << sum2(arr, n) << endl;
}
