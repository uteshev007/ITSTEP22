#include <iostream>
using namespace std;
// Передача аргумента по значению, т. е. делается копия значения и передаётся
void fn1(int a) { a = 1; }
// Передача по ссылке (существует в С++ и не существует в С)
void fn2(int &a) { a = 2; }
// Передача по указателю, благодаря разыменованию меняется ориганальное значение
void fn3(int *a) { *a = 3; }

/*
	Если перед звёздочкой стоит тип данных (любой), то создаём указатель
	Если после звёздочки идёт идёт адрес(например, с указателя), то достаём значение по адресу (оператор разыменования/косвенной адресации)
*/

int main() {
	int w = 0;
	fn1(w); // передача по значению, т. е. меняется копия
	cout << w << endl;
	fn2(w);	// передача по ссылке, т. е. меняется оригинал
	cout << w << endl;
	fn3(&w); // передача по указателю (передаётся копия адреса)
	cout << w << endl;



	cin.get();
}
