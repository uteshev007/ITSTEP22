#include <iostream>
#include <time.h>
using namespace std;

// Пример создания двумерного динамического массива

int main()
{
    setlocale(LC_ALL, "RUSSIAN");
    srand(time(nullptr));

    // Размеры для двумерного динамического массива
    int Rows = 10; // кол-во строк в массиве
    int Cols = 20; // кол-во столбцов в массиве

    // Создание двумерного динамического массива типа int размером [Row][Col]
    //  (выделение памяти под массив)

    // 1 шаг - Создаем массив указателей для строк.
    //   Длина массива массива укзателей == кол-ву строк Row
    // int ** Mas - указатель на массив указателей на строки (указатель на указатель типа int*)
    int ** Mas = new int*[Row];

    // 2 шаг - Создаем строки для двумерного массива
    for(int i=0; i < Row; i++)
    {
        // выделение памяти для i-ой строки - создание i-ой строки двумерного массива
        Mas[i] = new int [Col]; 
    }


    // тут работаем с динамическим двумерным массивом, как с обычным двумерным массивом
    for(int i=0; i < Row; i++) // цикл по строкам
    {
        for(int j=0; j < Col; j++) // цикл по столбцам
        {
            Mas[i][j] = rand() % 10;
            cout << Mas[i][j] << " ";
        }
        cout << endl; // конец вывода строки массива
    }


    // Удаление динамического массива, когда он становится не нужным,
    //   (освобождение памяти ранее выделеной памяти)
    // 1 шаг - удаление строк массива
    for(int i=0; i < Row; i++)
    {
        delete[] Mas[i]; // удаление i-ой строки
    }

    // 2 шаг - удаление массива указателей на строки
    delete[] Mas;

    cin.get(); cin.get();
}
