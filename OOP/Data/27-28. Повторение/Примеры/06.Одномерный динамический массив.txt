#include <iostream>
using namespace std;

int main()
{
	setlocale(LC_ALL, "RUSSIAN");
	int size = 1000; // размер для динамического массива

    // создание одномерного динамического массива типа int длиной size элементов,
    //  (выделение памяти под массив)
	int *mas = new int[size];

    // тут должен быть рабочий код с динамическим массивом
	cin.get();

    // удаление динамического массива, когда он становится не нужным,
    //   (освобождение памяти ранее выделеной памяти)
	delete[]mas;

	cin.get(); cin.get();
}
