/* примеры работы с сылкой и указателем */
struct Test
{
  int a;
  string name;
};
// глобальная ф-ция для печати переменных структуры Test
void Print(Test obj) {
  cout << "a = " << obj.a << endl;
  cout << "name = " << obj.name << endl;
}
// глобальная ф-ция для чтения переменных структуры Test
//void Input(Test obj) // передача по значению, как копию
void Input(Test & obj) // передача по ссылке, оригинал
{
  cout << "Enter a: ";
  (cin >> obj.a).ignore();
  cout << "Enter name: ";
  getline(cin, obj.name);
}
// перегруженный вариант Input
void Input(Test * obj) // передача по указателю, оригинал
{
  cout << "Enter a: ";
  (cin >> obj->a).ignore();
  cout << "Enter name: ";
  getline(cin, obj->name);
}

void func(int& a) {
  a = rand();
}
int main()
{
  Test A;
  //Input(A);
  //Print(A);
  A.a = 100;
  A.name = "Hello";
  Print(A);
  Input(A);    // Input(Test & obj)
  Print(A);
  Input( &A ); // Input(Test * obj)
  Print(A);
}
