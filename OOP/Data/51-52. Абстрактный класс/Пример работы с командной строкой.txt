/*
* Работа с командной строкой
*/
/*
  int    argc   - кол-во аргументов (arguments count)
  char * argv[] - массив строк аргументов (argument values)
  char * env[]  - переменные окружения (enveroment)
*/
//int main()
int main(int argc, char* argv[], char* env[])
{
  // кол-во аргументов
  cout << "argc: " << argc << endl;
  // список аргументов
  cout << "Arguments: \n";
  for (int i = 0; i < argc; i++) {
    cout << i << ") "
      << "\"" << argv[i] << "\"" << endl;
  }
  cout << "-------------------\n";
  // char *env[] - список строк переменных окружения программы
  //  Enveroment
  int i = 0;
  // Цикл вывода переменных окружения
  while (env[i] != nullptr)
  {
    cout << i << ") \"" << env[i] << "\"" << endl;
    i++;
  }
  cout << "Lenght of enveroment = " << i << endl;
}
