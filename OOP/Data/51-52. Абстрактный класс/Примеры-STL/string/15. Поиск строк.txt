#include <iostream>
#include <string>
using namespace std;
/*
29.	***Пользователь вводит 2 строки,
программа формирует третью строку, в которую помещает символы первой строки, не встречающиеся во второй строке.
*/

int main() {
	string s1, s2, s3;

	cout << "Enter your 1 string: ";
	getline(cin, s1);
	cout << endl;

	cout << "Enter your 2 string: ";
	getline(cin, s2);
	cout << endl;

	size_t found = 0;
	// Проходимся по первой строке
	for (int i = 0; i < s1.size(); i++) {
		found = s1.find_first_not_of(&s2.c_str()[found],i);
		if (found != string::npos) {
			s3.push_back(s1[found]);
		}
		//s3.push_back(s1[s1.find_first_not_of(s2[i], i)]);
	}

	cout << "Simbols that are not in s2: " << s3;

	cin.get();
	cin.get();
}
