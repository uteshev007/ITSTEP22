#include <iterator>
#include <iostream>

using namespace std;
int main()
{
	char a[] = { 'a','b','c' };
	//вывод данных на экран с использованием итератора вывода
	//в конструктор передается поток, в который надо
	//осуществлять вывод и разделитель 

	//без разделителя
	ostream_iterator<char> out(cout);
